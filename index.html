function isValidUrl(string) { try { new URL(string); return true; } catch (_) { return false; } } function toggleChatbot() { const window = document.getElementById('chatbotWindow'); chatbotOpen = !chatbotOpen; window.style.display = chatbotOpen ? 'flex' : 'none'; if (chatbotOpen) { document.getElementById('chatInput').focus(); } } function handleKeyPress(event) { if (event.key === 'Enter' && !isWaitingForResponse) { sendMessage(); } } function addMessage(message, isUser = false, isLoading = false) { const messagesContainer = document.getElementById('chatbotMessages'); const messageDiv = document.createElement('div'); if (isLoading) { messageDiv.className = 'message loading'; messageDiv.innerHTML = `
NovaBot está escribiendo... `; } else { messageDiv.className = `message ${isUser ? 'user' : 'bot'}`; messageDiv.textContent = message; } messagesContainer.appendChild(messageDiv); messagesContainer.scrollTop = messagesContainer.scrollHeight; return messageDiv; } function removeMessage(messageElement) { if (messageElement && messageElement.parentNode) { messageElement.parentNode.removeChild(messageElement); } } function addToCart(product) { cart.push(product); const confirmationMessage = `✅ ${product} ha sido añadido al carrito. Total de productos: ${cart.length}`; if (!chatbotOpen) { toggleChatbot(); // Esperar un poco para que se abra el chat antes de mostrar el mensaje setTimeout(() => { addMessage(confirmationMessage, false); }, 300); } else { addMessage(confirmationMessage, false); } } async function sendMessage() { const input = document.getElementById('chatInput'); const sendButton = document.getElementById('sendButton'); const message = input.value.trim(); if (!message || isWaitingForResponse) return; const webhookUrl = webhookUrlInput.value.trim(); if (!webhookUrl || !isValidUrl(webhookUrl)) { addMessage('⚠️ Error: Debes configurar la URL del webhook de n8n primero.', false); return; } // Deshabilitar input mientras procesamos isWaitingForResponse = true; input.disabled = true; sendButton.disabled = true; // Mostrar mensaje del usuario addMessage(message, true); input.value = ''; // Mostrar indicador de carga const loadingMessage = addMessage('', false, true); try { // Preparar datos para enviar al webhook const requestData = { message: message, cart: cart, cartCount: cart.length, timestamp: new Date().toISOString(), sessionId: generateSessionId() }; // Realizar petición al webhook de n8n const response = await fetch(webhookUrl, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify(requestData) }); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); // Remover indicador de carga removeMessage(loadingMessage); // Mostrar respuesta del bot const botResponse = data.response || data.message || 'Lo siento, no pude procesar tu mensaje correctamente.'; addMessage(botResponse, false); } catch (error) { console.error('Error al comunicarse con el webhook:', error); // Remover indicador de carga removeMessage(loadingMessage); // Mostrar mensaje de error let errorMessage = '❌ Error de conexión: '; if (error.name === 'TypeError' && error.message.includes('fetch')) { errorMessage += 'No se pudo conectar con el webhook. Verifica que n8n esté ejecutándose y que la URL sea correcta.'; } else if (error.message.includes('HTTP error')) { errorMessage += `El servidor respondió con error ${error.message.split('status: ')[1]}`; } else { errorMessage += 'Problema de comunicación con el asistente IA. Revisa la consola para más detalles.'; } addMessage(errorMessage, false); } finally { // Rehabilitar input isWaitingForResponse = false; input.disabled = false; sendButton.disabled = false; input.focus(); } } function generateSessionId() { return 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9); } // Smooth scrolling for navigation document.querySelectorAll('a[href^="#"]').forEach(anchor => { anchor.addEventListener('click', function (e) { e.preventDefault(); const target = document.querySelector(this.getAttribute('href')); if (target) { target.scrollIntoView({ behavior: 'smooth', block: 'start' }); } }); }); // Auto-open chatbot on page load after 3 seconds setTimeout(() => { if (!chatbotOpen) { toggleChatbot(); } }, 3000); // Cargar URL del webhook desde localStorage si existe document.addEventListener('DOMContentLoaded', function() { const savedWebhookUrl = localStorage.getItem('novashop_webhook_url'); if (savedWebhookUrl) { webhookUrlInput.value = savedWebhookUrl; webhookUrlInput.dispatchEvent(new Event('input')); } }); // Guardar URL del webhook en localStorage cuando cambie webhookUrlInput.addEventListener('change', function() { localStorage.setItem('novashop_webhook_url', this.value); });
